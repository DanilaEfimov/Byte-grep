cmake_minimum_required(VERSION 3.16)

project(byte-grep VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(Boost REQUIRED COMPONENTS program_options)

set(SOURCES
    src/main.cpp
    src/Byte.cpp
    src/Command.cpp
    src/CountCommand.cpp
    src/Counter.cpp
    src/MaskReader.cpp
    src/Pattern.cpp
    src/SearchCommand.cpp
    src/Selector.cpp
    src/Translator.cpp
)

set(HEADERS
    include/Byte.h
    include/Command.h
    include/CountCommand.h
    include/Counter.h
    include/MaskReader.h
    include/Pattern.h
    include/SearchCommand.h
    include/Selector.h
    include/Translator.h
    include/common.h
)

add_executable(byte-grep ${SOURCES} ${HEADERS})

target_include_directories(byte-grep PRIVATE include)

target_link_libraries(byte-grep PRIVATE Boost::program_options)

if(MSVC)
    target_compile_options(byte-grep PRIVATE /W4 /utf-8)
else()
    target_compile_options(byte-grep PRIVATE -Wall -Wextra -pedantic)
endif()
